Basics

  • p1. Packages
    • 要点: Goはpackage単位で構成。mainパッケージから実行開始。パッケージ名=インポートパスの最後の要素
package main

// パッケージをインポート
import "fmt"
import "math/rand"

func main() {
    // 規約で、パッケージ名はインポートパスの最後の要素と同じ名前に（rand）
    fmt.Println("My favorite number is", rand.Intn(10))
}
  • p2. Imports
    • 要点: importは括弧でグループ化（factored import）するのが推奨スタイル
package main

// インポートグループ化し( factored import statement )
import (
    "fmt"
    "math"
)

func main() {
    fmt.Printf("Now you have %g problems.\n", math.Sqrt(7))
}
  • p3. Exported names
    • 要点: 大文字始まり=公開（exported）、小文字始まり=非公開。他パッケージからは大文字のみアクセス可
package main

import (
    "fmt"
    "math"
)

func main() {
    // 大文字で始まる名前は、他のパッケージからアクセスできる（exported name）
    fmt.Println(math.Pi)
}
  • p4. Functions
    • 要点: 関数定義の基本形。Goでは変数名の後ろに型を書く（x int）
package main

import "fmt"

// cf. 型宣言の意向詳細 https://go.dev/blog/declaration-syntax
func add(x int, y int) int {
    return x + y
}

func main() {
    fmt.Println(add(42, 13))
}
  • p5. Functions continued
    • 要点: 同じ型の引数が連続する場合、最後の型名だけ残して省略できる
package main

import "fmt"

func add(x, y int) int { //x int, y int を x, y int へ省略
    return x + y
}

func main() {
    fmt.Println(add(42, 13))
}
  • p6. Multiple results
    • 要点: 関数は複数の戻り値を返せる。Goでは多値返却が一般的（エラー処理等で多用）
package main

import "fmt"

func swap(x, y string) (string, string) { // 引数: x, y 戻り値: 2つのstring
    return y, x  // 順序を逆にして返す
}

func main() {
    a, b := swap("hello", "world")
    // := 短縮変数宣言
    //    var a string
    //    var b string
    //    a, b = swap("hello", "world")

	fmt.Println(a, b)
}
  • p7. Named return values
    • 要点: 戻り値に名前をつけると関数先頭で変数定義扱い。returnだけで返せる（naked return）。短い関数でのみ推奨
package main

import "fmt"

func split(sum int) (x, y int) {  // 戻り値に名前(x, y)がついている
    x = sum * 4 / 9
    y = sum - x
    return  // これが naked return（値を指定していない）
}

func main() {
	fmt.Println(split(17))
}
  • p8. Variables
    • 要点: varで変数宣言。複数変数をまとめて宣言可能。パッケージレベルでも関数内でも使える
package main

import "fmt"

// 変数リスト宣言
var c, python, java bool // 複数の変数の最後に型

func main() {
	var i int
	fmt.Println(i, c, python, java)
}
  • p9. Variables with initializers
    • 要点: 初期化子（初期値）を与えれば型を省略可能。右辺の値から型が決まる
package main

import "fmt"

// 変数毎に初期化子( initializer )を指定
var i, j int = 1, 2

func main() {
	var c, python, java = true, false, "no!"
	fmt.Println(i, j, c, python, java)
}
  • p10. Short variable declarations
    • 要点: :=でvar+型を省略した短縮宣言。関数内でのみ使用可能（パッケージレベルでは不可）
package main

import "fmt"

func main() {
	var i, j int = 1, 2
	k := 3 // 型宣言を省略して、初期化子から型を推論（暗黙的な型宣言）
	c, python, java := true, false, "no!"

	fmt.Println(i, j, k, c, python, java)
}

// 1 2 3 true false no!
  • p11. Basic types
    • 要点: Goの組み込み型一覧。int/uintのサイズは環境依存(32 or 64bit)。特に理由がなければintを使う
package main

import (
    "fmt"
    "math/cmplx"
)

var (
    ToBe   bool       = false
    MaxInt uint64     = 1<<64 - 1
    z      complex128 = cmplx.Sqrt(-5 + 12i)
)

func main() {
    fmt.Printf("Type: %T Value: %v\n", ToBe, ToBe)
    fmt.Printf("Type: %T Value: %v\n", MaxInt, MaxInt)
    fmt.Printf("Type: %T Value: %v\n", z, z)
}

  • Goのビルトイン型：

    • interface{} (any)
    • bool
    • byte (uint8 の別名)
    • complex64 complex128
    • error
    • float32 float64
    • int int8 int16 int32 int64
    • rune string
    • uint uint8 uint16 uint32 uint64
    • uintptr

  • p12. Zero values

    • 要点: 初期値なし変数は自動でゼロ値に。数値→0、bool→false、string→""。未初期化=不定値にならない安全設計
package main

import "fmt"

// ゼロ値( zero value ):初期値無し変数はゼロ値で初期化
func main() {
    var i int      // 0
    var f float64  // 0
    var b bool     // false
    var s string   // ""

    fmt.Printf("%v %v %v %q\n", i, f, b, s)
}
// 0 0 false ""
  • p13. Type conversions
    • 要点: Goでは型変換は常に明示的。T(v)で変数vをT型に変換。暗黙変換は一切ない（C言語との大きな違い）
package main

import (
	"fmt"
	"math"
)

func main() {
    var x, y int = 3, 4

    var f float64 = math.Sqrt(float64(x*x + y*y))
    var z uint = uint(f)

    // 引数の変換関数を削除したら
    // var f float64 = math.Sqrt(x*x + y*y)   // float64() を削除 → エラー
    // var z uint = f                         // uint() を削除 → エラー

    // 変数宣言と同時に型変換を行うこともできる
    // f := math.Sqrt(float64(x*x + y*y))
    // z := uint(f)

    fmt.Println(x, y, z)
}

[!NOTE]
C言語では暗黙的に型変換（implicit conversion）が行われますが、
Goではすべての型変換を明示的に書く必要がある。
これにより、意図しない精度の損失やデータの切り捨てを防ぐ（コードの安全性UP）

  • p14. Type inference
    • 要点: :=やvar =で型を省略すると右辺から型推論される。リテラル42→int、3.14→float64、0.5+0.5i→complex128
package main

import "fmt"

func main() {
	v := 42 // change me!
	fmt.Printf("v is of type %T\n", v)
}
  • p15. Constants
    • 要点: constで定数宣言。文字・文字列・bool・数値のみ。:=では宣言不可。実行時に変更されない値に使う
package main

import "fmt"

const Pi = 3.14

func main() {
	const World = "世界"
	fmt.Println("Hello", World)
	fmt.Println("Happy", Pi, "Day")

	const Truth = true
	fmt.Println("Go rules?", Truth)
}
  • p16. Numeric Constants
    • 要点: 数値定数は高精度で型を持たない。使われる文脈に応じてintやfloat64に自動的に型が決まる
package main

import "fmt"

const (
	// Create a huge number by shifting a 1 bit left 100 places.
	// In other words, the binary number that is 1 followed by 100 zeroes.
	Big = 1 << 100
	// Shift it right again 99 places, so we end up with 1<<1, or 2.
	Small = Big >> 99
)

func needInt(x int) int { return x*10 + 1 }
func needFloat(x float64) float64 {
	return x * 0.1
}

func main() {
	fmt.Println(needInt(Small))
	fmt.Println(needFloat(Small))
	fmt.Println(needFloat(Big))
}

Flow control statements

  • p1. For
    • 要点: Goのforは()不要・{}必須。初期化; 条件; 後処理の3部構成。Goにはforしかループがない
package main

import "fmt"

func main() {
	sum := 0
	for i := 0; i < 10; i++ {
		sum += i
	}
	fmt.Println(sum)
}
  • p2. For continued
    • 要点: 初期化と後処理は省略可能。条件式だけ残す書き方もできる
package main

import "fmt"

func main() {
	sum := 1
	for ; sum < 1000; {
		sum += sum
	}
	fmt.Println(sum)
}
  • p3. For is Go’s “while”
    • 要点: セミコロンも省略するとwhile相当になる。Goにwhileキーワードはない
package main

import "fmt"

func main() {
	sum := 1
	for sum < 1000 {
		sum += sum
	}
	fmt.Println(sum)
}
  • p4. Forever
    • 要点: for {}で無限ループ。条件を全て省略した形
package main

func main() {
	for {
	}
}
  • p5. If
    • 要点: forと同じく()不要・{}必須
package main

import (
	"fmt"
	"math"
)

func sqrt(x float64) string {
	if x < 0 {
		return sqrt(-x) + "i"
	}
	return fmt.Sprint(math.Sqrt(x))
}

func main() {
	fmt.Println(sqrt(2), sqrt(-4))
}
  • p6. If with a short statement
    • 要点: ifの条件前に短いステートメントを書ける。そこで宣言した変数はifブロック内のみ有効
package main

import (
	"fmt"
	"math"
)

func pow(x, n, lim float64) float64 {
	if v := math.Pow(x, n); v < lim {
		return v
	}
	return lim
}

func main() {
	fmt.Println(
		pow(3, 2, 10),
		pow(3, 3, 20),
	)
}
  • p7. If and else
    • 要点: ifで宣言した変数はelseブロックでも使える。ただしif-elseの外では使えない
package main

import (
	"fmt"
	"math"
)

func pow(x, n, lim float64) float64 {
	if v := math.Pow(x, n); v < lim {
		return v
	} else {
		fmt.Printf("%g >= %g\n", v, lim)
	}
	// can't use v here, though
	return lim
}

func main() {
	fmt.Println(
		pow(3, 2, 10),
		pow(3, 3, 20),
	)
}

package main

import (
	"fmt"

  • p9. Switch
    • 要点: Goのswitchは自動break（C言語のようにfallthroughしない）。caseは定数でなくてもよい
package main

import (
	"fmt"
	"runtime"
)

func main() {
	fmt.Print("Go runs on ")
	switch os := runtime.GOOS; os {
	case "darwin":
		fmt.Println("OS X.")
	case "linux":
		fmt.Println("Linux.")
	default:
		// freebsd, openbsd,
		// plan9, windows...
		fmt.Printf("%s.\n", os)
	}
}
  • p10. Switch evaluation order
    • 要点: caseは上から順に評価され、一致した時点で停止。それ以降のcaseは評価されない
package main

import (
	"fmt"
	"time"
)

func main() {
	fmt.Println("When's Saturday?")
	today := time.Now().Weekday()
	switch time.Saturday {
	case today + 0:
		fmt.Println("Today.")
	case today + 1:
		fmt.Println("Tomorrow.")
	case today + 2:
		fmt.Println("In two days.")
	default:
		fmt.Println("Too far away.")
	}
}
  • p11. Switch with no condition
    • 要点: 条件なしswitch = switch true。長いif-else if-elseチェーンの代替として使える
package main

import (
	"fmt"
	"time"
)

func main() {
	t := time.Now()
	switch {
	case t.Hour() < 12:
		fmt.Println("Good morning!")
	case t.Hour() < 17:
		fmt.Println("Good afternoon.")
	default:
		fmt.Println("Good evening.")
	}
}
  • p12. Defer
    • 要点: deferは関数のreturn時まで実行を遅延。引数はdefer時に即時評価される。リソース解放等に便利
package main

import "fmt"

func main() {
	defer fmt.Println("world")

	fmt.Println("hello")
}
  • p13. Stacking defers
    • 要点: 複数のdeferはスタック（LIFO: 後入れ先出し）順で実行される
package main

import "fmt"

func main() {
	fmt.Println("counting")

	for i := 0; i < 10; i++ {
		defer fmt.Println(i)
	}

	fmt.Println("done")
}

More types

  • p1. Pointers
    • 要点: &でアドレス取得、*で参照先の値を読み書き（デリファレンス）。C言語と違いポインタ演算は不可
package main

import "fmt"

func main() {
	i, j := 42, 2701

	p := &i         // point to i
	fmt.Println(*p) // read i through the pointer
	*p = 21         // set i through the pointer
	fmt.Println(i)  // see the new value of i

	p = &j         // point to j
	*p = *p / 37   // divide j through the pointer
	fmt.Println(j) // see the new value of j
}
  • p2. Structs
    • 要点: structはフィールドの集まり。Goにクラスはなく、structが基本のデータ構造
package main

import "fmt"

type Vertex struct {
	X int
	Y int
}

func main() {
	fmt.Println(Vertex{1, 2})
}
  • p3. Struct Fields
    • 要点: .（ドット）でフィールドにアクセス
package main

import "fmt"

type Vertex struct {
	X int
	Y int
}

func main() {
	v := Vertex{1, 2}
	v.X = 4
	fmt.Println(v.X)
}
  • p4. Pointers to structs
    • 要点: ポインタ経由でもp.Xでアクセス可能。(*p).Xと書く必要はない（Goが自動解決）
package main

import "fmt"

type Vertex struct {
	X int
	Y int
}

func main() {
	v := Vertex{1, 2}
	p := &v
	p.X = 1e9
	fmt.Println(v)
}
  • p5. Struct Literals
    • 要点: structリテラルで初期化。Name:で一部フィールドだけ指定可。&でポインタを直接取得
package main

import "fmt"

type Vertex struct {
	X, Y int
}

var (
	v1 = Vertex{1, 2}  // has type Vertex
	v2 = Vertex{X: 1}  // Y:0 is implicit
	v3 = Vertex{}      // X:0 and Y:0
	p  = &Vertex{1, 2} // has type *Vertex
)

func main() {
	fmt.Println(v1, p, v2, v3)
}
  • p6. Arrays
    • 要点: [n]Tで固定長配列。サイズは型の一部なので変更不可。実用ではスライスの方が一般的
package main

import "fmt"

func main() {
	var a [2]string
	a[0] = "Hello"
	a[1] = "World"
	fmt.Println(a[0], a[1])
	fmt.Println(a)

	primes := [6]int{2, 3, 5, 7, 11, 13}
	fmt.Println(primes)
}
  • p7. Slices
    • 要点: []Tで可変長のスライス。a[low:high]で配列の一部を参照（halfopen: lowは含む、highは含まない）
package main

import "fmt"

func main() {
	primes := [6]int{2, 3, 5, 7, 11, 13}

	var s []int = primes[1:4]
	fmt.Println(s)
}
  • p8. Slices are like references to arrays
    • 要点: スライスはデータのコピーではなく元配列への参照。要素を変更すると元配列と他スライスにも反映される
package main

import "fmt"

func main() {
	names := [4]string{
		"John",
		"Paul",
		"George",
		"Ringo",
	}
	fmt.Println(names)

	a := names[0:2]
	b := names[1:3]
	fmt.Println(a, b)

	b[0] = "XXX"
	fmt.Println(a, b)
	fmt.Println(names)
}
  • p9. Slice literals
    • 要点: []T{値...}でスライスリテラル（長さ指定なしの配列リテラル）。配列を作りつつそれを参照するスライスが返る
package main

import "fmt"

func main() {
	q := []int{2, 3, 5, 7, 11, 13}
	fmt.Println(q)

	r := []bool{true, false, true, true, false, true}
	fmt.Println(r)

	s := []struct {
		i int
		b bool
	}{
		{2, true},
		{3, false},
		{5, true},
		{7, true},
		{11, false},
		{13, true},
	}
	fmt.Println(s)
}
  • p10. Slice defaults
    • 要点: 上下限の既定値は下限=0、上限=len。a[:]で全体、a[:3]で先頭3つ、a[2:]で2番目以降
package main

import "fmt"

func main() {
	s := []int{2, 3, 5, 7, 11, 13}

	s = s[1:4]
	fmt.Println(s)

	s = s[:2]
	fmt.Println(s)

	s = s[1:]
	fmt.Println(s)
}
  • p11. Slice length and capacity
    • 要点: len(s)=現在の要素数、cap(s)=元配列の先頭からの容量。容量内なら再スライスで長さを伸ばせる
package main

import "fmt"

func main() {
	s := []int{2, 3, 5, 7, 11, 13}
	printSlice(s)

	// Slice the slice to give it zero length.
	s = s[:0]
	printSlice(s)

	// Extend its length.
	s = s[:4]
	printSlice(s)

	// Drop its first two values.
	s = s[2:]
	printSlice(s)
}

func printSlice(s []int) {
	fmt.Printf("len=%d cap=%d %v\n", len(s), cap(s), s)
}
  • p12. Nil slices
    • 要点: スライスのゼロ値はnil。長さ・容量ともに0で元配列も持たない
package main

import "fmt"

func main() {
	var s []int
	fmt.Println(s, len(s), cap(s))
	if s == nil {
		fmt.Println("nil!")
	}
}
  • p13. Creating a slice with make
    • 要点: make([]T, len, cap)で動的サイズのスライスを生成。ゼロ値で初期化された配列を内部に持つ
package main

import "fmt"

func main() {
	a := make([]int, 5)
	printSlice("a", a)

	b := make([]int, 0, 5)
	printSlice("b", b)

	c := b[:2]
	printSlice("c", c)

	d := c[2:5]
	printSlice("d", d)
}

func printSlice(s string, x []int) {
	fmt.Printf("%s len=%d cap=%d %v\n",
		s, len(x), cap(x), x)
}
  • p14. Slices of slices
    • 要点: スライスの要素にスライスを持てる（二次元配列的な使い方）
package main

import (
	"fmt"
	"strings"
)

func main() {
	// Create a tic-tac-toe board.
	board := [][]string{
		[]string{"_", "_", "_"},
		[]string{"_", "_", "_"},
		[]string{"_", "_", "_"},
	}

	// The players take turns.
	board[0][0] = "X"
	board[2][2] = "O"
	board[1][2] = "X"
	board[1][0] = "O"
	board[0][2] = "X"

	for i := 0; i < len(board); i++ {
		fmt.Printf("%s\n", strings.Join(board[i], " "))
	}
}
  • p15. Appending to a slice
    • 要点: append(s, 値...)で要素追加。容量を超えると自動で新しい配列を確保して拡張される
package main

import "fmt"

func main() {
	var s []int
	printSlice(s)

	// append works on nil slices.
	s = append(s, 0)
	printSlice(s)

	// The slice grows as needed.
	s = append(s, 1)
	printSlice(s)

	// We can add more than one element at a time.
	s = append(s, 2, 3, 4)
	printSlice(s)
}

func printSlice(s []int) {
	fmt.Printf("len=%d cap=%d %v\n", len(s), cap(s), s)
}
  • p16. Range
    • 要点: for i, v := rangeでスライス/mapを反復。2値（index, 値のコピー）を返す
package main

import "fmt"

var pow = []int{1, 2, 4, 8, 16, 32, 64, 128}

func main() {
	for i, v := range pow {
		fmt.Printf("2**%d = %d\n", i, v)
	}
}
  • p17. Range continued
    • 要点: _で不要な値を捨てる。indexだけ必要なら2番目の変数自体を省略可能
package main

import "fmt"

func main() {
	pow := make([]int, 10)
	for i := range pow {
		pow[i] = 1 << uint(i) // == 2**i
	}
	for _, value := range pow {
		fmt.Printf("%d\n", value)
	}

// Inc increments the counter for the given key.
func (c *SafeCounter) Inc(key string) {
	c.mu.Lock()
	// Lock so only one goroutine at a time can access the map c.v.
	c.v[key]++
	c.mu.Unlock()
}

// Value returns the current value of the counter for the given key.
func (c *SafeCounter) Value(key string) int {
	c.mu.Lock()
	// Lock so only one goroutine at a time can access the map c.v.
	defer c.mu.Unlock()
	return c.v[key]
}

func main() {
	c := SafeCounter{v: make(map[string]int)}
	for i := 0; i < 1000; i++ {
		go c.Inc("somekey")
	}

	time.Sleep(time.Second)
	fmt.Println(c.Value("somekey"))
}
